% This file contains the modifications to the nusthesis.cls template

\usepackage{xfrac}

% Package for circuits
\usepackage{circuitikz}
%Specifies how thick the bipoles are, e.g. the capacitor, the inductor, ...
\ctikzset{bipoles/thickness=1.5}

% Automatic abbreviations page
\input{acronym.tex}

% Disable word hyphenation breaks
% \usepackage[none]{hyphenat}
\tolerance=1
\emergencystretch=.2\maxdimen
\hyphenpenalty=10000
\hbadness=10000

% Reduce vertical spacing between chapter title
\titlespacing*{\chapter}{0pt}{-12.5ex}{-3ex}

% Reduce vertical spacing between equations
\AtBeginDocument{
  \addtolength{\abovedisplayskip}{-2.5ex}
  \addtolength{\abovedisplayshortskip}{-2.5ex}
  \addtolength{\belowdisplayskip}{0ex}
  \addtolength{\belowdisplayshortskip}{0ex}
}

% Redefine ABSTRACT
% Change title to be same as chapter 
\def\abstract{
  \intendclearpage
  \phantomsection\addcontentsline{toc}{chapter}{\abstractname}
  \begin{alwayssingle}
  \chapter*{\abstractname}
  \begin{center}{
  }
  \vspace{.1in}
  \end{center}
}

% Redefine ACKNOWLEDGEMENTS
% Change title to be same as chapter
\renewenvironment{acknowledgments}
  {\intendclearpage
   \phantomsection\addcontentsline{toc}{chapter}{\acknowledgename}
   \begin{alwayssingle}
   \chapter*{\acknowledgename}
   \begin{center}
   \end{center}
  }
  {\end{alwayssingle}}

% Define LIST OF SYMBOLS AND ABBREVIATIONS
\def\abbreviationsname{List of Abbreviations}

\newenvironment{abbreviations}
{\intendclearpage
 \phantomsection\addcontentsline{toc}{chapter}{\abbreviationsname}
 \begin{alwayssingle}
 \chapter*{\abbreviationsname}
 \begin{center}
 \end{center}
}
{\end{alwayssingle}}

%% Cross-Referencing

% Capitalise Chapter and Section for \autoref
\AtBeginDocument{\def\chapterautorefname{Chapter}}
\AtBeginDocument{\def\sectionautorefname{Section}}

% Define \fullref with one single link
\newcommand*{\fullref}[1]{\hyperref[{#1}]{\autoref*{#1} \nameref*{#1}}}
